var MyClass = function (x, y) {
    "use strict";
    
    this.first = x;
    this.second = y;
};

MyClass.prototype.sayHi = function () {
  setTimeout(function () {
    $("#result").html(this.first + " " + this.second);
  }, 1);
};

var c = new MyClass("Hello", "World");
c.sayHi();
